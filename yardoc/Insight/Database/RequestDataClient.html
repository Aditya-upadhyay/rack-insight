def store(env, *keys_and_value)
  return if env.nil?
  request_id = env["insight.request-id"]
  return if request_id.nil?

  value = keys_and_value[-1]
  keys = keys_and_value[0...-1]

  @table.store(request_id, value, @key_sql_template % keys)
end
