def initialize(table_name, *keys)
  @table_name = table_name
  @keys = keys
  if(execute("select * from sqlite_master where name = ?", table_name).empty?)
    logger.warn{ "Initializing a table called #{table_name}" }
    execute(create_sql)
  end
end
