def queries_to_param

  end

end

def record_call(method, time, hit, key)
  if Array === key
    @queries << Query.new(:get_multi, time, hit, key)
  else
    @queries << Query.new(method.to_sym, time, hit, [key])
  end
  @calls += 1
  @time += time
  @keys += keys
end
